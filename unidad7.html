void setup() {
  // Configurar pines
  pinMode(boton1, INPUT_PULLUP);
  pinMode(boton2, INPUT_PULLUP);

  } else {

  // Monitoreo por serial

  digitalWrite(led2, LOW);
  
  // LED 3 parpadea
